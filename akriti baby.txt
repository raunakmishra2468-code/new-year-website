    <script>
        /**
         * CONFIGURATION
         * Customize the experience here.
         */
        const CONFIG = {
            recipientName: "Darling", // The name used in the typing text
            senderName: "Your Secret Admirer", // Your name
            // The typing text in Phase 2
            reflectionText: [
                "The world moves fast...",
                "But some moments stand still.",
                "Thank you for being my highlight of 2025."
            ],
            // The final letter in Phase 4
            finalMessage: `As the clock strikes midnight, I don't wish for a perfect year.
            
            I simply wish for a year filled with *us*. 
            
            More laughter, more adventures, and more quiet moments.
            You are the future I look forward to.
            
            Here's to 2026.`
        };

        /**
         * APP LOGIC
         */
        const App = {
            stage: 1,
            elements: {
                ui: document.getElementById('ui-layer'),
                stages: [
                    document.getElementById('stage-1'),
                    document.getElementById('stage-2'),
                    document.getElementById('stage-3'),
                    document.getElementById('stage-4')
                ],
                typewriter: document.getElementById('typewriter-text'),
                countdown: document.getElementById('countdown'),
                finalMsg: document.getElementById('final-message'),
                sender: document.getElementById('sender-name')
            },

            init() {
                // Fade in UI on load
                setTimeout(() => {
                    this.elements.ui.style.opacity = 1;
                }, 500);
                
                // Initialize Canvases
                Starfield.init();
                Fireworks.init();
                
                // Set text
                this.elements.finalMsg.innerText = CONFIG.finalMessage;
                this.elements.sender.innerText = CONFIG.senderName;
            },

            nextStage() {
                this.elements.stages[this.stage - 1].classList.remove('active');
                this.stage++;
                const next = this.elements.stages[this.stage - 1];
                
                if(next) {
                    next.classList.add('active');
                    this.handleStageLogic();
                }
            },

            handleStageLogic() {
                if (this.stage === 2) {
                    this.runTypewriter();
                } else if (this.stage === 3) {
                    this.runCountdown();
                } else if (this.stage === 4) {
                    Fireworks.startCelebration();
                }
            },

            async runTypewriter() {
                const textElement = this.elements.typewriter;
                const lines = CONFIG.reflectionText;

                for (let line of lines) {
                    textElement.innerText = "";
                    await this.typeLine(line, textElement);
                    await new Promise(r => setTimeout(r, 1500)); // Pause reading
                }
                
                // Transition to next stage automatically after reading
                this.nextStage();
            },

            typeLine(text, element) {
                return new Promise(resolve => {
                    let i = 0;
                    const interval = setInterval(() => {
                        element.innerText += text.charAt(i);
                        i++;
                        if (i > text.length - 1) {
                            clearInterval(interval);
                            resolve();
                        }
                    }, 50); // Typing speed
                });
            },

            runCountdown() {
                let count = 3;
                const el = this.elements.countdown;
                
                const timer = setInterval(() => {
                    count--;
                    if (count > 0) {
                        el.innerText = count;
                        // Add a little pop effect
                        el.style.transform = "scale(1.2)";
                        setTimeout(() => el.style.transform = "scale(1)", 200);
                    } else {
                        clearInterval(timer);
                        this.nextStage();
                    }
                }, 1000);
            }
        };

        /**
         * VISUALS: BACKGROUND STARFIELD
         */
        const Starfield = {
            canvas: document.getElementById('star-canvas'),
            ctx: null,
            stars: [],
            
            init() {
                this.ctx = this.canvas.getContext('2d');
                this.resize();
                window.addEventListener('resize', () => this.resize());
                
                for(let i=0; i<150; i++) {
                    this.stars.push({
                        x: Math.random() * this.canvas.width,
                        y: Math.random() * this.canvas.height,
                        size: Math.random() * 2,
                        speed: Math.random() * 0.2 + 0.05
                    });
                }
                this.animate();
            },

            resize() {
                this.canvas.width = window.innerWidth;
                this.canvas.height = window.innerHeight;
            },

            animate() {
                this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                this.ctx.fillStyle = "white";
                
                this.stars.forEach(star => {
                    this.ctx.globalAlpha = Math.random() * 0.5 + 0.3;
                    this.ctx.beginPath();
                    this.ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2);
                    this.ctx.fill();
                    
                    // Move upwards slowly
                    star.y -= star.speed;
                    if(star.y < 0) star.y = this.canvas.height;
                });

                requestAnimationFrame(() => this.animate());
            }
        };

        /**
         * VISUALS: SOPHISTICATED FIREWORKS
         */
        const Fireworks = {
            canvas: document.getElementById('firework-canvas'),
            ctx: null,
            particles: [],
            rockets: [],
            isCelebrating: false,

            init() {
                this.ctx = this.canvas.getContext('2d');
                this.resize();
                window.addEventListener('resize', () => this.resize());
                this.loop();
            },

            resize() {
                this.canvas.width = window.innerWidth;
                this.canvas.height = window.innerHeight;
            },

            startCelebration() {
                this.isCelebrating = true;
                // Launch initial barrage
                for(let i=0; i<5; i++) {
                    setTimeout(() => this.launchRocket(), i * 300);
                }
                // Continuous random launches
                setInterval(() => {
                    if(document.hidden) return;
                    this.launchRocket();
                }, 800);
            },

            launchRocket() {
                const x = Math.random() * this.canvas.width * 0.8 + (this.canvas.width * 0.1);
                const targetY = Math.random() * this.canvas.height * 0.4 + (this.canvas.height * 0.1);
                
                this.rockets.push({
                    x: x,
                    y: this.canvas.height,
                    targetY: targetY,
                    speed: Math.random() * 3 + 8,
                    color: `hsl(${Math.random() * 360}, 100%, 70%)`
                });
            },

            explode(x, y, color) {
                const particleCount = 80;
                for(let i=0; i<particleCount; i++) {
                    const angle = (Math.PI * 2) / particleCount * i;
                    const velocity = Math.random() * 6 + 2;
                    this.particles.push({
                        x: x,
                        y: y,
                        vx: Math.cos(angle) * velocity,
                        vy: Math.sin(angle) * velocity,
                        alpha: 1,
                        color: color,
                        decay: Math.random() * 0.015 + 0.01
                    });
                }
            },

            loop() {
                // Clear with trail effect
                this.ctx.globalCompositeOperation = 'destination-out';
                this.ctx.fillStyle = 'rgba(0, 0, 0, 0.1)'; // Trails
                this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
                this.ctx.globalCompositeOperation = 'lighter';

                // Update Rockets
                for(let i = this.rockets.length - 1; i >= 0; i--) {
                    const r = this.rockets[i];
                    r.y -= r.speed;
                    r.speed *= 0.98; // Friction

                    this.ctx.fillStyle = r.color;
                    this.ctx.beginPath();
                    this.ctx.arc(r.x, r.y, 3, 0, Math.PI * 2);
                    this.ctx.fill();

                    if(r.y <= r.targetY || r.speed < 1) {
                        this.explode(r.x, r.y, r.color);
                        this.rockets.splice(i, 1);
                    }
                }

                // Update Particles
                for(let i = this.particles.length - 1; i >= 0; i--) {
                    const p = this.particles[i];
                    p.x += p.vx;
                    p.y += p.vy;
                    p.vy += 0.05; // Gravity
                    p.alpha -= p.decay;

                    if(p.alpha <= 0) {
                        this.particles.splice(i, 1);
                        continue;
                    }

                    this.ctx.globalAlpha = p.alpha;
                    this.ctx.fillStyle = p.color;
                    this.ctx.beginPath();
                    this.ctx.arc(p.x, p.y, 2, 0, Math.PI * 2);
                    this.ctx.fill();
                }

                requestAnimationFrame(() => this.loop());
            }
        };

        // Start everything
        window.onload = () => App.init();

    </script>
